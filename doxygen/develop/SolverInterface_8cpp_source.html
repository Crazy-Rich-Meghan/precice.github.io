10 std::string_view toSV(precice::string_view sv)
11 {
12  // The given string_view may contain null chars.
13  // We trim to the first null char here.
14  std::string_view s{sv.data(), sv.size()};
15  auto trim_pos = s.find('\0');
16  if (trim_pos != s.npos) {
17  s.remove_suffix(s.size() - trim_pos);
18  }
19  return s;
20 }
